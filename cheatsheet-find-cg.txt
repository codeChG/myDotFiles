


# finde alle Dateien rekursiv,
# kopiere sie und beim kopieren nicht überschreiben sondern backups 
#
find /srcdir -name "*.txt*" ! -name ".*" -type f -exec cp -rv --backup=numbered {} /destdir \;

# setzt fdupes voraus
# findet alle Duplikate (md5) im aktuellen Verzeichnis und löscht diese, 
# erhält nur das erste Original (order)
#
fdupes --noprompt --delete --order="name" .


# Suche mit regex
# finde alle pdf files mit ._ im Pfad
# fish-kompatibel ($ am Ende auskommentiert):
#
find /home/christian/muttibackup/ -regex "^.*\/\.\_.*\.pdf\$"


# finde alle (versteckten) Directories die mit . beginnen
# und loesche sie
#
find . -regextype sed -regex "^\.\/\..*" -type d -exec rm -rf {} \+
find . -regextype sed -regex "^\.\/\..*" -type d -exec rm -rf "{}" \+
#
# mit -depth 
# listet zuerst Dateien, dann Verzeichnisse!
# ohne -depth Fehlermeldung, wenn Verzeichnisse nicht leer,
# löscht also zuerst die Dateien, dann die geleerten Verzeichnisse,
# mit -print zeigt geloeschte Datei(en) und Verzeichnisse
#
find . -depth -regextype sed -regex "^\.\/\..*" -print -delete
